useEffect

useEffect maneja efectos secundarios como llamadas a API, suscripciones, timers, etc.

import { useState, useEffect } from 'react';

function EjemplosUseEffect() {
  const [count, setCount] = useState(0);
  const [users, setUsers] = useState([]);

  // 1. Sin dependencias - se ejecuta en cada render
  useEffect(() => {
    console.log('Se ejecuta en cada render');
  });

  // 2. Array vac√≠o - solo se ejecuta una vez (al montar)
  useEffect(() => {
    console.log('Componente montado');
    
    // Simular llamada a API
    fetch('/api/users')
      .then(res => res.json())
      .then(data => setUsers(data));
      
    // Cleanup opcional
    return () => {
      console.log('Componente se va a desmontar');
    };
  }, []); // Array de dependencias vac√≠o

  // 3. Con dependencias - se ejecuta cuando cambia 'count'
  useEffect(() => {
    document.title = `Contador: ${count}`;
  }, [count]); // Se ejecuta cuando 'count' cambia

  // 4. M√∫ltiples useEffect para diferentes responsabilidades
  useEffect(() => {
    const timer = setInterval(() => {
      console.log('Timer ejecut√°ndose...');
    }, 1000);

    // Limpiar el timer cuando el componente se desmonte
    return () => clearInterval(timer);
  }, []);

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>
        Incrementar
      </button>
      <ul>
        {users.map(user => (
          <li key={user.id}>{user.name}</li>
        ))}
      </ul>
    </div>
  );
}

useContext

useContext permite consumir valores de un contexto sin prop drilling.

import { createContext, useContext, useState } from 'react';

// 1. Crear el contexto
const ThemeContext = createContext();

// 2. Crear el Provider
function ThemeProvider({ children }) {
  const [theme, setTheme] = useState('light');
  
  const toggleTheme = () => {
    setTheme(prevTheme => prevTheme === 'light' ? 'dark' : 'light');
  };

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

// 3. Hook personalizado para usar el contexto
function useTheme() {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme debe usarse dentro de ThemeProvider');
  }
  return context;
}

// 4. Componente que consume el contexto
function Header() {
  const { theme, toggleTheme } = useTheme();

  return (
    <header className={`header ${theme}`}>
      <h1>Mi Aplicaci√≥n</h1>
      <button onClick={toggleTheme}>
        Cambiar a {theme === 'light' ? 'dark' : 'light'}
      </button>
    </header>
  );
}

// 5. Uso en la aplicaci√≥n
function App() {
  return (
    <ThemeProvider>
      <Header />
      <main>Contenido de la aplicaci√≥n</main>
    </ThemeProvider>
  );
}

En los formularios controlados, React controla el valor de los inputs a trav√©s del estado.

import { useState } from 'react';

function FormularioContacto() {
  const [formData, setFormData] = useState({
    nombre: '',
    email: '',
    mensaje: '',
    acepta: false
  });

  const [errores, setErrores] = useState({});

  const handleChange = (e) => {
    const { name, value, type, checked } = e.target;
    
    setFormData(prevData => ({
      ...prevData,
      [name]: type === 'checkbox' ? checked : value
    }));

    // Limpiar error cuando el usuario empieza a escribir
    if (errores[name]) {
      setErrores(prev => ({ ...prev, [name]: '' }));
    }
  };

  const validarFormulario = () => {
    const nuevosErrores = {};

    if (!formData.nombre.trim()) {
      nuevosErrores.nombre = 'El nombre es requerido';
    }

    if (!formData.email) {
      nuevosErrores.email = 'El email es requerido';
    } else if (!/\S+@\S+\.\S+/.test(formData.email)) {
      nuevosErrores.email = 'Email inv√°lido';
    }

    if (!formData.mensaje.trim()) {
      nuevosErrores.mensaje = 'El mensaje es requerido';
    }

    if (!formData.acepta) {
      nuevosErrores.acepta = 'Debes aceptar los t√©rminos';
    }

    setErrores(nuevosErrores);
    return Object.keys(nuevosErrores).length === 0;
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    
    if (validarFormulario()) {
      console.log('Formulario v√°lido:', formData);
      // Aqu√≠ enviar√≠as los datos
      alert('Formulario enviado!');
      
      // Reset del formulario
      setFormData({
        nombre: '',
        email: '',
        mensaje: '',
        acepta: false
      });
    }
  };

  return (
    <form onSubmit={handleSubmit} className="max-w-md mx-auto">
      <div className="mb-4">
        <label className="block mb-2">Nombre:</label>
        <input
          type="text"
          name="nombre"
          value={formData.nombre}
          onChange={handleChange}
          className={`w-full p-2 border rounded ${
            errores.nombre ? 'border-red-500' : 'border-gray-300'
          }`}
        />
        {errores.nombre && (
          <span className="text-red-500 text-sm">{errores.nombre}</span>
        )}
      </div>

      <div className="mb-4">
        <label className="block mb-2">Email:</label>
        <input
          type="email"
          name="email"
          value={formData.email}
          onChange={handleChange}
          className={`w-full p-2 border rounded ${
            errores.email ? 'border-red-500' : 'border-gray-300'
          }`}
        />
        {errores.email && (
          <span className="text-red-500 text-sm">{errores.email}</span>
        )}
      </div>

      <div className="mb-4">
        <label className="block mb-2">Mensaje:</label>
        <textarea
          name="mensaje"
          value={formData.mensaje}
          onChange={handleChange}
          rows="4"
          className={`w-full p-2 border rounded ${
            errores.mensaje ? 'border-red-500' : 'border-gray-300'
          }`}
        />
        {errores.mensaje && (
          <span className="text-red-500 text-sm">{errores.mensaje}</span>
        )}
      </div>

      <div className="mb-4">
        <label className="flex items-center">
          <input
            type="checkbox"
            name="acepta"
            checked={formData.acepta}
            onChange={handleChange}
            className="mr-2"
          />
          Acepto los t√©rminos y condiciones
        </label>
        {errores.acepta && (
          <span className="text-red-500 text-sm">{errores.acepta}</span>
        )}
      </div>

      <button
        type="submit"
        className="w-full bg-blue-500 text-white p-2 rounded hover:bg-blue-600"
      >
        Enviar
      </button>
    </form>
  );
}

Hook Personalizado para Formularios

Crear un hook reutilizable para manejar formularios.

// Hook personalizado para formularios
function useForm(valoresIniciales, validaciones) {
  const [valores, setValores] = useState(valoresIniciales);
  const [errores, setErrores] = useState({});
  const [tocado, setTocado] = useState({});

  const handleChange = (e) => {
    const { name, value, type, checked } = e.target;
    const nuevoValor = type === 'checkbox' ? checked : value;

    setValores(prev => ({
      ...prev,
      [name]: nuevoValor
    }));

    // Validar campo si ya fue tocado
    if (tocado[name] && validaciones[name]) {
      const error = validaciones[name](nuevoValor);
      setErrores(prev => ({
        ...prev,
        [name]: error
      }));
    }
  };

  const handleBlur = (e) => {
    const { name } = e.target;
    setTocado(prev => ({ ...prev, [name]: true }));

    if (validaciones[name]) {
      const error = validaciones[name](valores[name]);
      setErrores(prev => ({
        ...prev,
        [name]: error
      }));
    }
  };

  const validarFormulario = () => {
    const nuevosErrores = {};
    let esValido = true;

    Object.keys(validaciones).forEach(campo => {
      const error = validaciones[campo](valores[campo]);
      if (error) {
        nuevosErrores[campo] = error;
        esValido = false;
      }
    });

    setErrores(nuevosErrores);
    setTocado(
      Object.keys(validaciones).reduce((acc, campo) => {
        acc[campo] = true;
        return acc;
      }, {})
    );

    return esValido;
  };

  const reset = () => {
    setValores(valoresIniciales);
    setErrores({});
    setTocado({});
  };

  return {
    valores,
    errores,
    tocado,
    handleChange,
    handleBlur,
    validarFormulario,
    reset
  };
}

function FormularioArchivos() {
  const [archivo, setArchivo] = useState(null);
  const [preview, setPreview] = useState(null);

  const handleFileChange = (e) => {
    const file = e.target.files[0];
    setArchivo(file);

    // Crear preview para im√°genes
    if (file && file.type.startsWith('image/')) {
      const reader = new FileReader();
      reader.onloadend = () => {
        setPreview(reader.result);
      };
      reader.readAsDataURL(file);
    } else {
      setPreview(null);
    }
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    
    if (!archivo) {
      alert('Selecciona un archivo');
      return;
    }

    const formData = new FormData();
    formData.append('archivo', archivo);
    formData.append('descripcion', 'Mi archivo');

    try {
      const response = await fetch('/api/upload', {
        method: 'POST',
        body: formData
      });

      if (response.ok) {
        alert('Archivo subido exitosamente');
        setArchivo(null);
        setPreview(null);
      }
    } catch (error) {
      console.error('Error subiendo archivo:', error);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label>Seleccionar archivo:</label>
        <input
          type="file"
          onChange={handleFileChange}
          accept="image/*,.pdf,.doc,.docx"
        />
      </div>

      {preview && (
        <div>
          <img 
            src={preview} 
            alt="Preview" 
            style={{ maxWidth: '200px', maxHeight: '200px' }}
          />
        </div>
      )}

      {archivo && (
        <div>
          <p>Archivo seleccionado: {archivo.name}</p>
          <p>Tama√±o: {(archivo.size / 1024).toFixed(2)} KB</p>
        </div>
      )}

      <button type="submit" disabled={!archivo}>
        Subir Archivo
      </button>
    </form>
  );
}
