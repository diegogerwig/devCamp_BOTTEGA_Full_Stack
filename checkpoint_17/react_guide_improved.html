Formularios en React

Aprende a manejar formularios de manera eficiente con componentes controlados y estado local

Select Dropdowns en React

Los elementos <select> en React funcionan como componentes controlados. El valor se maneja a través del estado del componente, lo que permite un control total sobre la selección del usuario. 

import React, { useState } from 'react';

function SelectExample() {
  const [selectedOption, setSelectedOption] = useState('');
  
  const handleChange = (event) => {
    setSelectedOption(event.target.value);
  };
  
  return (
    <div>
      <label>Elige una opción:</label>
      <select value={selectedOption} onChange={handleChange}>
        <option value="">-- Selecciona --</option>
        <option value="react">React</option>
        <option value="vue">Vue</option>
        <option value="angular">Angular</option>
      </select>
      <p>Seleccionaste: {selectedOption}</p>
    </div>
  );
}

Punto clave: Siempre usa la prop value junto con onChange para mantener el componente controlado y sincronizado con el estado.

Integración Avanzada de Dropzone

Una implementación más completa que incluye validación de archivos, previsualización de imágenes y manejo de errores. 

function AdvancedDropzone() {
  const [uploadedFiles, setUploadedFiles] = useState([]);
  const [errors, setErrors] = useState([]);
  
  const { getRootProps, getInputProps } = useDropzone({
    accept: {
      'image/*': []
    },
    maxSize: 5242880, // 5MB
    onDrop: (acceptedFiles, rejectedFiles) => {
      // Procesar archivos aceptados
      const filesWithPreview = acceptedFiles.map(file => 
        Object.assign(file, {
          preview: URL.createObjectURL(file)
        })
      );
      
      setUploadedFiles(prev => [...prev, ...filesWithPreview]);
      
      // Manejar archivos rechazados
      if (rejectedFiles.length > 0) {
        const newErrors = rejectedFiles.map(file => 
          `${file.file.name}: ${file.errors[0].message}`
        );
        setErrors(newErrors);
      }
    }
  });
  
  const removeFile = (index) => {
    setUploadedFiles(files => files.filter((_, i) => i !== index));
  };
  
  return (
    <div>
      <div {...getRootProps()} className="dropzone">
        <input {...getInputProps()} />
        <p>Arrastra imágenes (máx. 5MB) o haz clic</p>
      </div>
      
      {errors.length > 0 && (
        <div className="errors">
          {errors.map((error, index) => (
            <p key={index} style={{ color: 'red' }}>{error}</p>
          ))}
        </div>
      )}
      
      <div className="preview-container">
        {uploadedFiles.map((file, index) => (
          <div key={index} className="preview-item">
            <img src={file.preview} alt={file.name} width={100} />
            <p>{file.name}</p>
            <button onClick={() => removeFile(index)}>Eliminar</button>
          </div>
        ))}
      </div>
    </div>
  );
}

Upload de Miniaturas

Crear miniaturas automáticamente al subir imágenes mejora el rendimiento y la experiencia del usuario en aplicaciones web. 

function ThumbnailUploader() {
  const [thumbnails, setThumbnails] = useState([]);
  
  const createThumbnail = (file) => {
    return new Promise((resolve) => {
      const canvas = document.createElement('canvas');
      const ctx = canvas.getContext('2d');
      const img = new Image();
      
      img.onload = () => {
        // Redimensionar a 150x150
        canvas.width = 150;
        canvas.height = 150;
        
        ctx.drawImage(img, 0, 0, 150, 150);
        
        canvas.toBlob((blob) => {
          resolve({
            original: file,
            thumbnail: blob,
            preview: URL.createObjectURL(blob)
          });
        });
      };
      
      img.src = URL.createObjectURL(file);
    });
  };
  
  const { getRootProps, getInputProps } = useDropzone({
    accept: { 'image/*': [] },
    onDrop: async (acceptedFiles) => {
      const thumbnailPromises = acceptedFiles.map(createThumbnail);
      const results = await Promise.all(thumbnailPromises);
      setThumbnails(prev => [...prev, ...results]);
    }
  });
  
  return (
    <div>
      <div {...getRootProps()}>
        <input {...getInputProps()} />
        <p>Sube imágenes para crear miniaturas</p>
      </div>
      
      <div style={{ display: 'flex', flexWrap: 'wrap' }}>
        {thumbnails.map((item, index) => (
          <div key={index} style={{ margin: '10px' }}>
            <img src={item.preview} alt="thumbnail" />
            <p>{item.original.name}</p>
          </div>
        ))}
      </div>
    </div>
  );
}

function BannerLogoUploader() {

  const [logo, setLogo] = useState(null);
  
  const createDropzone = (type, setter, dimensions) => {
    return useDropzone({
      accept: { 'image/*': [] },
      maxFiles: 1,
      onDrop: (acceptedFiles) => {
        const file = acceptedFiles[0];
        if (file) {
          // Validar dimensiones si es necesario
          const img = new Image();
          img.onload = () => {
            if (dimensions) {

React Refs - Acceso Directo al DOM

Los refs permiten acceder directamente a elementos DOM o componentes. Úsalos cuando necesites enfocar inputs, medir elementos o integrar librerías externas. 

import { useRef, useEffect } from 'react';

function FocusInput() {
  const inputRef = useRef(null);
  
  // Enfocar automáticamente al montar
  useEffect(() => {
    inputRef.current.focus();
  }, []);
  
  const handleFocus = () => {
    inputRef.current.focus();
  };
  
  const handleClear = () => {
    inputRef.current.value = '';
    inputRef.current.focus();
  };
  
  return (
    <div>
      <input 
        ref={inputRef}
        type="text"
        placeholder="Escribe algo..."
      />
      <button onClick={handleFocus}>Enfocar</button>
      <button onClick={handleClear}>Limpiar</button>
    </div>
  );
}

// Ref para medir elementos
function MeasureElement() {
  const divRef = useRef(null);
  const [dimensions, setDimensions] = useState(null);
  
  const measureElement = () => {
    if (divRef.current) {
      const rect = divRef.current.getBoundingClientRect();
      setDimensions({
        width: rect.width,
        height: rect.height
      });
    }
  };
  
  return (
    <div>
      <div ref={divRef} style={{ padding: '20px', background: 'lightblue' }}>
        Elemento a medir
      </div>
      <button onClick={measureElement}>Medir</button>
      {dimensions && (
        <p>Dimensiones: {dimensions.width}px x {dimensions.height}px</p>
      )}
    </div>
  );
}

Mixin para Inputs de Formulario

Un hook personalizado que simplifica el manejo de múltiples inputs en formularios, reduciendo código repetitivo y mejorando la mantenibilidad. 

// Hook personalizado para formularios
function useFormInput(initialValue) {
  const [value, setValue] = useState(initialValue);
  
  const handleChange = (e) => {
    setValue(e.target.value);
  };
  
  const reset = () => {
    setValue(initialValue);
  };
  
  return {
    value,
    onChange: handleChange,
    reset
  };
}

// Uso del hook en un formulario
function ContactForm() {

}

Manejador de Eliminación

Implementar eliminación segura con confirmación del usuario y manejo de errores. Siempre proporciona feedback visual y permite deshacer acciones críticas. 

function ItemList() {

    { id: 1, name: 'Producto A', price: 100 },

    { id: 3, name: 'Producto C', price: 150 }

  const [deletingId, setDeletingId] = useState(null);

    // Confirmar eliminación

  return (

        <div key={item.id} style={{

          margin: '5px 0'

    const newErrors = {};

    }
    
    if (!data.price || data.price <= 0) {
      newErrors.price = 'El precio debe ser mayor a 0';
    }

// Hook para consultas dinámicas con debouncing
function useApiQuery(baseUrl, initialParams = {}) {
  const [data, setData] = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);
  const [params, setParams] = useState(initialParams);
  
  // Debounce para búsquedas
  const [debouncedParams, setDebouncedParams] = useState(params);
  
  useEffect(() => {
    const timer = setTimeout(() => {
      setDebouncedParams(params);
    }, 300);
    
    return () => clearTimeout(timer);
  }, [params]);
  
  // Realizar consulta cuando cambien los parámetros
  useEffect(() => {
    const fetchData = async () => {
      setLoading(true);
      setError(null);
      
      try {
        const queryString = new URLSearchParams(debouncedParams).toString();
        const url = `${baseUrl}?${queryString}`;
        
        const response = await fetch(url);
        
        if (!response.ok) {
          throw new Error(`Error ${response.status}: ${response.statusText}`);
        }
        
        const result = await response.json();
        setData(result.data || result);
        
      } catch (err) {
        setError(err.message);
        setData([]);
      } finally {
        setLoading(false);
      }
    };
    
    fetchData();
  }, [baseUrl, debouncedParams]);
  
  const updateParams = (newParams) => {
    setParams(prev => ({ ...prev, ...newParams }));
  };
  
  const resetParams = () => {
    setParams(initialParams);
  };
  
  return { data, loading, error, updateParams, resetParams, params };
}

function ImageEditor({ initialImages, onImagesChange }) {
  const [images, setImages] = useState(initialImages || []);
  const [previewUrls, setPreviewUrls] = useState(new Map());
  const [isUploading, setIsUploading] = useState(false);
  
  // Limpiar URLs de objeto al desmontar (evitar memory leaks)
  useEffect(() => {
    return () => {
      previewUrls.forEach(url => {
        if (url.startsWith('blob:')) {
          URL.revokeObjectURL(url);
        }
      });
    };
  }, [previewUrls]);
  
  // Sincronizar con props cuando cambien
  useEffect(() => {
    if (initialImages && initialImages !== images) {
      setImages(initialImages);
    }
  }, [initialImages]);
  
  const addImages = (newFiles) => {
    const newImages = [];
    const newPreviews = new Map(previewUrls);
    
    newFiles.forEach((file, index) => {
      const tempId = `temp-${Date.now()}-${index}`;
      const previewUrl = URL.createObjectURL(file);
      
      newImages.push({
        id: tempId,
        file,
        name: file.name,
        size: file.size,
        isNew: true,
        isUploading: false
      });
      
      newPreviews.set(tempId, previewUrl);
    });
    
    setImages(prev => [...prev, ...newImages]);
    setPreviewUrls(newPreviews);
  };
  
  const removeImage = (imageId) => {
    // Limpiar URL de preview si existe
    const previewUrl = previewUrls.get(imageId);
    if (previewUrl && previewUrl.startsWith('blob:')) {
      URL.revokeObjectURL(previewUrl);
    }
    
    // Actualizar estados
    setImages(prev => prev.filter(img => img.id !== imageId));
    setPreviewUrls(prev => {
      const newMap = new Map(prev);
      newMap.delete(imageId);
      return newMap;
    });
    
    // Notificar cambio al componente padre
    const updatedImages = images.filter(img => img.id !== imageId);
    onImagesChange?.(updatedImages);
  };
  
  const uploadImage = async (imageId) => {
    const image = images.find(img => img.id === imageId);
    if (!image || !image.isNew) return;
    
    // Marcar como subiendo
    setImages(prev => prev.map(img => 
      img.id === imageId 
        ? { ...img, isUploading: true, error: null }
        : img
    ));
    
    try {
      const formData = new FormData();
      formData.append('image', image.file);
      
      const response = await fetch('/api/upload', {
        method: 'POST',
        body: formData
      });
      
      if (!response.ok) {
        throw new Error(`Upload failed: ${response.statusText}`);
      }
      
      const result = await response.json();
      
      // Actualizar imagen con datos del servidor
      setImages(prev => prev.map(img => 
        img.id === imageId 
          ? { 
              ...img, 
              id: result.id,
              url: result.url,
              isNew: false,
              isUploading: false,
              file: null // Limpiar archivo para liberar memoria
            }
          : img
      ));
      
      // Limpiar preview URL temporal
      const previewUrl = previewUrls.get(imageId);
      if (previewUrl) {
        URL.revokeObjectURL(previewUrl);
        setPreviewUrls(prev => {
          const newMap = new Map(prev);
          newMap.delete(imageId);
          newMap.set(result.id, result.url);
          return newMap;
        });
      }
      
    } catch (error) {
      console.error('Upload error:', error);
      
      // Marcar error en la imagen
      setImages(prev => prev.map(img => 
        img.id === imageId 
          ? { ...img, isUploading: false, error: error.message }
          : img
      ));
    }
  };
  
  const uploadAllImages = async () => {
    setIsUploading(true);

    try {
      await Promise.all(uploadPromises);
    } catch (error) {
      console.error('Batch upload error:', error);
    } finally {
      setIsUploading(false);
    }
  };

    if (image.url) return image.url;
    return previewUrls.get(image.id) || '';
  };

          type="file"
          multiple
          accept="image/*"

          style={{ marginRight: '10px' }}
        />

          disabled={isUploading || !images.some(img => img.isNew)}
          style={{ 
            backgroundColor: '#007bff',

        >

function useImageDeletion(apiEndpoint) {
  const [deletingIds, setDeletingIds] = useState(new Set());
  const [errors, setErrors] = useState(new Map());
  
  const deleteImage = async (imageId, optimisticUpdate) => {
    // Marcar como eliminando
    setDeletingIds(prev => new Set(prev).add(imageId));
    setErrors(prev => {
      const newErrors = new Map(prev);
      newErrors.delete(imageId);
      return newErrors;
    });
    
    // Actualización optimista
    const rollback = optimisticUpdate?.(imageId);
    
    try {
      const response = await fetch(`${apiEndpoint}/${imageId}`, {
        method: 'DELETE',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${localStorage.getItem('authToken')}`
        }
      });
      
      if (!response.ok) {
        const errorData = await response.json().catch(() => ({}));
        throw new Error(
          errorData.message || 
          `Error ${response.status}: ${response.statusText}`
        );
      }
      
      // Éxito - la actualización optimista se mantiene
      return true;
      
    } catch (error) {
      console.error('Delete error:', error);
      
      // Revertir actualización optimista
      rollback?.();
      
      // Guardar error
      setErrors(prev => new Map(prev).set(imageId, error.message));
      
      return false;
      
    } finally {
      // Quitar de la lista de eliminando
      setDeletingIds(prev => {
        const newSet = new Set(prev);
        newSet.delete(imageId);
        return newSet;
      });
    }
  };
  
  const clearError = (imageId) => {
    setErrors(prev => {
      const newErrors = new Map(prev);
      newErrors.delete(imageId);
      return newErrors;
    });
  };
  
  return {
    deleteImage,
    clearError,
    isDeleting: (id) => deletingIds.has(id),
    getError: (id) => errors.get(id),
    hasErrors: errors.size > 0
  };
}

      `¿Estás seguro de eliminar "${image?.name || 'esta imagen'}"?`
    );
    
    if (!confirmed) return;
    
    // Función de actualización optimista
    const optimisticUpdate = (id) => {
      const originalImages = [...images];
      
      // Marcar como eliminada (no remover completamente)
      setImages(prev => prev.map(img => 
        img.id === id 
          ? { ...img, isDeleted: true, deletedAt: new Date() }
          : img
      ));
      
      // Función de rollback
      return () => {
        setImages(originalImages);
      };

      // Notificar al componente padre
      const updatedImages = images.filter(img => img.id !== imageId);
