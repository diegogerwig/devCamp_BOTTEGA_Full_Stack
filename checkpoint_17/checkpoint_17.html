Formularios en React

Guide to Select Dropdowns in React

Los dropdowns en React se manejan con el estado para controlar el valor seleccionado.

import React, { useState } from 'react';

function SelectExample() {
  const [selected, setSelected] = useState('');
  
  return (
    <select 
      value={selected} 
      onChange={e => setSelected(e.target.value)}
      className="bg-dark-700 text-white p-2 rounded w-full"
    >
      <option value="">Selecciona una opción</option>
      <option value="1">Opción 1</option>
      <option value="2">Opción 2</option>
    </select>
  );
}

Guide to the Textarea Tag in React

El textarea en React se maneja similar a otros inputs, controlando su valor mediante estado.

function TextareaExample() {
  const [text, setText] = useState('');
  
  return (
    <textarea
      value={text}
      onChange={e => setText(e.target.value)}
      className="bg-dark-700 text-white p-2 rounded w-full h-32"
      placeholder="Escribe aquí..."
    />
  );
}

Populating Parent Component Data

Pasar datos de componentes hijos a padres usando funciones callback.

// Componente Padre
function Parent() {
  const [childData, setChildData] = useState(null);
  
  const handleChildData = (data) => {
    setChildData(data);
  };
  
  return (
    <Child onData={handleChildData} />
  );
}

// Componente Hijo
function Child({ onData }) {
  const [data, setData] = useState('');
  
  const sendToParent = () => {
    onData(data);
  };
  
  return (
    <div>
      <input value={data} onChange={e => setData(e.target.value)} />
      <button onClick={sendToParent}>Enviar</button>
    </div>
  );
}

Manejo de Imágenes

Installing React Dropzone Component

Instalación y configuración básica de React Dropzone para uploads de imágenes.

// Instalación
npm install react-dropzone

// Uso básico
import { useDropzone } from 'react-dropzone';

function DropzoneBasic() {
  const { getRootProps, getInputProps } = useDropzone({
    accept: {
      'image/*': []
    },
    onDrop: acceptedFiles => {
      console.log(acceptedFiles);
    }
  });

  return (
    <div {...getRootProps()} className="border-2 border-dashed border-gray-600 p-8 text-center cursor-pointer">
      <input {...getInputProps()} />
      <p>Arrastra imágenes aquí o haz clic</p>
    </div>
  );
}

Uploading Thumbnail Images to the API

Subir imágenes miniatura a una API usando Axios.

const uploadImage = async (file) => {
  const formData = new FormData();
  formData.append('thumbnail', file);
  
  try {
    const response = await axios.post('/api/upload', formData, {
      headers: {
        'Content-Type': 'multipart/form-data'
      }
    });
    return response.data;
  } catch (error) {
    console.error('Upload error:', error);
  }
};

Refs y Estilos

Overview of React Refs to Clear Form

Usar refs para acceder a elementos DOM y limpiar formularios.

import { useRef } from 'react';

function FormWithRef() {
  const formRef = useRef(null);
  
  const handleClear = () => {
    if (formRef.current) {
      formRef.current.reset();
    }
  };
  
  return (
    <form ref={formRef}>
      <!-- campos del formulario -->
      <button type="button" onClick={handleClear}>
        Limpiar
      </button>
    </form>
  );
}

Operaciones CRUD

Building a Delete Click Handler

Implementar un manejador de eliminación con confirmación.

const handleDelete = (id) => {
  if (window.confirm('¿Estás seguro de eliminar este elemento?')) {
    // Llamar a la API para eliminar
    axios.delete(`/api/items/${id}`)
      .then(() => {
        // Actualizar estado
        setItems(prevItems => prevItems.filter(item => item.id !== id));
      })
      .catch(error => {
        console.error('Delete error:', error);
      });
  }
};
